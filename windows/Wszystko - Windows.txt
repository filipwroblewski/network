1.	ZARZĄDZANIE DYSKAMI:

	diskpart:  [rozmiar podawany w MB]
	
		list disk/partition/volume/vdisk
		detail
		select
		create partition primary/extended/logical/EFI/MSR size
		create volume simple/stripe/mirror size=1000 disk1,2
		extend size=1000 disk2
		shrink desired=100      -zmniejszenei rozmiaru woluminu
		break disk
		active
		clean
		delete
		format fs=ntfs label="nazwa" quick
		assign letter
	
	convert N:/fs:ntfs			N-litera dysklu do konwersji z fat32 do ntfs
	
	chkdsk - informacje o dysku
	
2.	UŻYTKOWNICY:

	net user nazwa_uzytkownika /add
	net user nazwa_uzytkownika
	net user nazwa_uzytkownika haslo
	net user nazwa_uzytkownika /logonpasswordchg:yes
	net user nazwa_uzytkownika /passwordchg:yes
	net user nazwa_uzytkownika /times:wt-cz,09:00-15:00
	net user nazwa_uzytkownika /expires:2017-09-08
	net user nazwa_uzytkownika /full:"pełna nazwa"
